172 -(BOOL)removeFromGraph:(MKitModel *)model
173 {
174  BOOL exists=([modelStack objectForKey:model.modelId]!=nil);
175  if (exists)
176  [modelStack removeObjectForKey:model.modelId];
177 
178  if ((model.objectId!=nil) && ((id)model.objectId!=[NSNull null]))
179  {
180  NSString *className=NSStringFromClass([model class]);
181  NSMutableDictionary *objectCache=[classCache objectForKey:className];
182  if (!objectCache)
183  return exists;
184 
185  if ([objectCache objectForKey:model.objectId])
186  {
187  [objectCache removeObjectForKey:model.objectId];
188 
189  objectCount--;
190  size-=malloc_size(model);

192  return YES;
193  }
194  }
195 
196  return exists;

221 -(void)modelIdentifierChanged:(NSNotification *)notification
222 {
223  MKitModel *model=(MKitModel *)notification.object;
224 
225  if ([[model class] graph]!=self)
226  return;
227 
228  NSString *oldValue=[notification.userInfo objectForKey:NSKeyValueChangeOldKey];
229 
230  if ([modelStack objectForKey:oldValue])
231  [modelStack removeObjectForKey:oldValue];
232 
233  if ((model.modelId) && (![[self class] conformsToProtocol:@protocol(MKitNoGraph)]))
234  [modelStack setObject:model forKey:model.modelId];
