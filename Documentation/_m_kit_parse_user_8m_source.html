76 +(BOOL)logInWithUserName:(NSString *)userName password:(NSString *)password error:(NSError **)error
77 {
78  MKitParseServiceManager *manager=(MKitParseServiceManager *)[[self class] service];
79 
80  AFHTTPRequestOperation *op=[manager requestWithMethod:@"GET"
81  path:@"login"
82  params:@{@"username":userName,@"password":password}
83  body:nil
84  contentType:nil];
85 
86  [op start];
87  [op waitUntilFinished];
88 
89  if ([op hasAcceptableStatusCode])
90  {
91  NSDictionary *data=[op.responseString objectFromJSONString];
92  NSString *objId=data[@"objectId"];
93 
94  MKitParseUser *user=[MKitParseUser instanceWithObjectId:objId];
95  if (user.modelState==ModelStateNeedsData)
96  {
97  BOOL result=[user fetch:error];
98  if (!result)
99  {
100  [user removeFromGraph];
101  return NO;
102  }
103  }
104 
105  user.modelSessionToken=data[@"sessionToken"];
106  [manager storeUserCredentials:user];
107 
108  _currentUser=user;
109  [[NSNotificationCenter defaultCenter] postNotificationName:MKitUserLoggedInNotification object:_currentUser];
110 
111  return YES;
112  }
113 
114  if (error!=nil)
115  *error=op.error;
116 
117  return NO;
118 }
