82 -(void)executeInBackgroundWithLimit:(NSInteger)limit skip:(NSInteger)skip resultBlock:(MKitQueryResultBlock)resultBlock
83 {
84  dispatch_async(dispatch_get_global_queue(0, 0), ^{
85  NSError *error=nil;
86  NSDictionary *result=[self executeWithLimit:limit skip:skip error:&error];
87 
88  if ((result!=nil) && (resultBlock!=nil))
89  resultBlock(result[MKitQueryResultKey], [result[MKitQueryItemCountKey] integerValue], error);
90  else if (resultBlock)
91  resultBlock(nil, 0, error);
92  });
93 }
