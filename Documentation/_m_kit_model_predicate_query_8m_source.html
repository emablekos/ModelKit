29 -(NSPredicate *)buildQuery
30 {
31  NSMutableArray *predicates=[NSMutableArray array];
32 
33  for(NSDictionary *c in conditions)
34  {
35  id val=c[@"value"];
36  id val2=c[@"value2"];
37 
38  MKitGeoPoint *geoPoint=nil;
39  double distance=0.0;
40 
41  switch ([c[@"condition"] integerValue])
42  {
43  case KeyEquals:
44  [predicates addObject:[NSPredicate predicateWithFormat:[NSString stringWithFormat:@"%@ == %%@",c[@"key"]],val]];
45  break;
46  case KeyNotEqual:
47  [predicates addObject:[NSPredicate predicateWithFormat:[NSString stringWithFormat:@"%@ != %%@",c[@"key"]],val]];
48  break;
49  case KeyGreaterThanEqual:
50  [predicates addObject:[NSPredicate predicateWithFormat:[NSString stringWithFormat:@"%@ >= %%@",c[@"key"]],val]];
51  break;
52  case KeyGreaterThan:
53  [predicates addObject:[NSPredicate predicateWithFormat:[NSString stringWithFormat:@"%@ > %%@",c[@"key"]],val]];
54  break;
55  case KeyLessThan:
56  [predicates addObject:[NSPredicate predicateWithFormat:[NSString stringWithFormat:@"%@ < %%@",c[@"key"]],val]];
57  break;
58  case KeyLessThanEqual:
59  [predicates addObject:[NSPredicate predicateWithFormat:[NSString stringWithFormat:@"%@ <= %%@",c[@"key"]],val]];
60  break;
61  case KeyIn:
62  [predicates addObject:[NSPredicate predicateWithFormat:[NSString stringWithFormat:@"%@ IN %%@",c[@"key"]],val]];
63  break;
64  case KeyNotIn:
65  [predicates addObject:[NSPredicate predicateWithFormat:[NSString stringWithFormat:@"NOT (%@ IN %%@)",c[@"key"]],val]];
66  break;
67  case KeyExists:
68  [predicates addObject:[NSPredicate predicateWithFormat:[NSString stringWithFormat:@"%@ != NIL",c[@"key"]]]];
69  break;
70  case KeyNotExist:
71  [predicates addObject:[NSPredicate predicateWithFormat:[NSString stringWithFormat:@"%@ == NIL",c[@"key"]]]];
72  break;
73  case KeyWithin:
74  [predicates addObject:[NSPredicate predicateWithFormat:[NSString stringWithFormat:@"%@ >= %%@ AND %@ <= %%@",c[@"key"],c[@"key"]],val,val2]];
75  break;
76  case KeyBeginsWith:
77  [predicates addObject:[NSPredicate predicateWithFormat:[NSString stringWithFormat:@"%@ BEGINSWITH[cd] %%@",c[@"key"]],val]];
78  break;
79  case KeyEndsWith:
80  [predicates addObject:[NSPredicate predicateWithFormat:[NSString stringWithFormat:@"%@ ENDSWITH[cd] %%@",c[@"key"]],val]];
81  break;
82  case KeyLike:
83  [predicates addObject:[NSPredicate predicateWithFormat:[NSString stringWithFormat:@"%@ LIKE[cd] *%%@*",c[@"key"]],val]];
84  break;
85  case KeyWithinDistance:
86  geoPoint=[val objectForKey:@"point"];
87  distance=[[val objectForKey:@"distance"] doubleValue];
88  [predicates addObject:[NSPredicate predicateWithBlock:^BOOL(id evaluatedObject, NSDictionary *bindings) {
89  MKitGeoPoint *p=(MKitGeoPoint *)[evaluatedObject valueForKey:c[@"key"]];
90  double cdistance=acos(sin(p.latitudeRad) * sin(geoPoint.latitudeRad) + cos(p.latitudeRad) * cos(geoPoint.latitudeRad) * cos(geoPoint.longitudeRad - (p.longitudeRad))) * EARTH_RADIUS;
91  return (cdistance <= distance);
92  }]];
93  default:
94  break;
95  }
96  }
97 
98  return [NSCompoundPredicate andPredicateWithSubpredicates:predicates];
99 }
