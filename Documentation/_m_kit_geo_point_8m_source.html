68 -(MKitGeoBounds *)boundingCoordinatesForDistance:(double)distance
69 {
70  double radDist = distance / EARTH_RADIUS;
71 
72  double minLat = _latitudeRad - radDist;
73  double maxLat = _latitudeRad + radDist;
74 
75  double minLon, maxLon;
76  if ((minLat > MIN_LAT) && (maxLat < MAX_LAT))
77  {
78  double deltaLon = asin(sin(radDist) / cos(_latitudeRad));
79 
80  minLon = _longitudeRad - deltaLon;
81  if (minLon < MIN_LON)
82  minLon += 2.0 * M_PI;
83 
84  maxLon = _longitudeRad + deltaLon;
85  if (maxLon > MAX_LON)
86  maxLon -= 2.0 * M_PI;
87  }
88  else
89  {
90  // a pole is within the distance
91  minLat = MAX(minLat, MIN_LAT);
92  maxLat = MIN(maxLat, MAX_LAT);
93  minLon = MIN_LON;
94  maxLon = MAX_LON;
95  }
96 
97  return [MKitGeoBounds boundsWithMinPoint:[[self class] geoPointWithLatitude:RADIANS_TO_DEGREES(minLat) andLongitude:RADIANS_TO_DEGREES(minLon)]
98  maxPoint:[[self class] geoPointWithLatitude:RADIANS_TO_DEGREES(maxLat) andLongitude:RADIANS_TO_DEGREES(maxLon)]];
99 }
